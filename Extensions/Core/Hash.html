def stringify(options = {})
  preserve_nil = options.delete(:preserve_nil)||false
  
  inject({}) do |options, (key, value)|
    options[key.to_s] = (preserve_nil && value.nil? ? nil : value.to_s)
    options
  end
end

def stringify!(options = {})
  preserve_nil = options.delete(:preserve_nil)||false

  stringify_keys!
  each do |key, value|
    store(key, (preserve_nil && value.nil? ? nil : value.to_s))
  end
end
